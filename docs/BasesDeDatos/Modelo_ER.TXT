@startchen

left to right direction
skinparam shadowing false

entity Usuario {
  id_usuario <<key>>
  email
  pass
  nombre
  estado
  fecha_creacion
}

entity Rol {
  id_rol <<key>>
  tipo
}

entity Clase {
  id_clase <<key>>
  nombre
  descripcion
  nivel
}

entity Sesion {
  id_sesion <<key>>
  fecha
  hora_inicio
  hora_fin
  plazas_totales
}

entity Reserva {
  id_reserva <<key>>
  fecha_reserva
  estado
}

entity Usuario_Rol {
  id_usuario <<FK>>
  id_rol <<FK>>
}


relationship asigna_rol {
}

Usuario -1- asigna_rol
Usuario_Rol -N- asigna_rol

Rol -1- asigna_rol
Usuario_Rol -N- asigna_rol


relationship crea {
}

Usuario -1- crea
Clase -N- crea


relationship imparte {
}

Usuario -1- imparte
Sesion -N- imparte


relationship pertenece {
}

Clase -1- pertenece
Sesion -N- pertenece


relationship realiza {
}

Usuario -1- realiza
Reserva -N- realiza


relationship genera {
}

Sesion -1- genera
Reserva -N- genera

@endchen
